Notation cat_of obj :=
  (@Build_PreCategory obj
                      (fun x y ⇒ x → y)
                      (fun _ x ⇒ x)
                      (fun _ _ _ f g ⇒ f o g)%core
                      (fun _ _ _ _ _ _ _ ⇒ idpath)
                      (fun _ _ _ ⇒ idpath)
                      (fun _ _ _ ⇒ idpath)
                      _).
